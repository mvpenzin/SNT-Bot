UserManager
При инициализации проверить таблицу:
CREATE TABLE IF NOT EXISTS snt_users (
created TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'Дата создания пользователя',
modified TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT 'Дата обновления пользователя',
user_id BIGINT UNSIGNED NOT NULL PRIMARY KEY COMMENT 'Уникальный идентификатор пользователя',
user_name VARCHAR(64) NOT NULL COMMENT 'Наименование пользователя',
user_fio VARCHAR(255) COMMENT 'Ф.И.О. пользователя',
user_phone VARCHAR(10) COMMENT 'Телефон пользователя',
comment TEXT COMMENT 'Комментарий',
INDEX (user_name)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
При входе пользователя по start вызывается user_add:
INSERT INTO snt_users (user_id, user_name)
VALUES (%s, %s)
ON DUPLICATE KEY UPDATE user_name = VALUES(user_name)
При редактировании пользователя будет вызываться user_fio_edt:
UPDATE snt_users
SET (user_fio)
VALUES (%s)
WHERE user_id = user_id
При редактировании пользователя будет вызываться user_phone_edt:
UPDATE snt_users
SET (user_phone)
VALUES (%s)
WHERE id = id
Запрос на вывод информации о пользователе - user_info - вывести user_id, user_name, user_fio, user_phone.

DetailsManager
При инициализации проверить таблицу:
CREATE TABLE IF NOT EXISTS snt_deteils (
created TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'Дата создания записи',
modified TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT 'Дата обновления записи',
id VARCHAR(8) NOT NULL PRIMARY KEY COMMENT 'Идентификатор',
name VARCHAR(120) NOT NULL COMMENT 'Наименование получателя',
inn NUMBER(10) NOT NULL COMMENT 'ИНН получателя',
kpp NUMBER(9) NOT NULL COMMENT 'КПП получателя',
personal_acc NUMBER(20) NOT NULL COMMENT 'Счёт получателя',
bank_name VARCHAR(120) NOT NULL COMMENT 'Наименование банка',
bik NUMBER(9) NOT NULL COMMENT 'БИК',
corresp_acc NUMBER(20) NOT NULL COMMENT 'Корреспондентский счёт',
comment TEXT COMMENT 'Комментарий',
UNIQ INDEX (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
Предусмотреть вывод всех реквизитов в текст - details_text, в QR - details_qr, а так же в состав ссылки - details_url.

ContactsManager
При инициализации проверить таблицу:
CREATE TABLE IF NOT EXISTS snt_contacts (
created TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'Дата создания записи',
modified TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT 'Дата обновления записи',
prior NUMBER(3) NOT NULL COMMENT 'Приоритет контакта',
type VARCHAR(20) NOT NULL PRIMARY KEY COMMENT 'Тип контакта',
value VARCHAR(120) NOT NULL COMMENT 'Значение контакта',
adds VARCHAR(240) COMMENT 'Дополнительные сведения',
comment TEXT COMMENT 'Комментарий',
UNIQ INDEX (type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
